import java.util.*;

class Solution {

   private static boolean[][] player;

   public int solution(int n, int[][] results) {

      int answer = 0;

      player = new boolean[n + 1][n + 1];

      for(int i = 1; i <= n; i++) {
         player[i][i] = true;
      }

      for(int[] result : results) {
         int winner = result[0];
         int loser = result[1];

         player[winner][loser] = true;
      }

      for(int i = 1; i <= n; i++) {
         for(int j = 1; j <= n; j++) {
            if(player[j][i]) {
               for(int z = 1; z <= n; z++) {
                  if(player[i][z]) player[j][z] = true;
               }
            }
         }
      }

      for(int i = 1; i <= n; i++) {
         boolean rank = true;
         for(int j = 1; j <= n; j++) {
            if(i == j) continue;
            if(!player[i][j] && !player[j][i]) {
               rank = false;
               break;
            }
         }
         if(rank) answer++;
      }

      return answer;
   }

}
