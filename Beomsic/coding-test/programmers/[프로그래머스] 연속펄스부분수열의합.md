class Solution {

  public long solution(int[] sequence) {

    long answer = 0;
    int length = sequence.length;

    long[][] dp = new long[2][length];

    dp[0][0] = sequence[0];
    dp[1][0] = sequence[0] * -1;
    answer = Math.max(dp[0][0], dp[1][0]);

    int type = -1;
    for(int i = 1; i < length; i++) {

      // 1 -1 1... ìˆœìœ¼ë¡œ ì§„í–‰ ì¤‘ì¸ ê²½ìš°ì˜ í˜„ì¬ í„ìŠ¤ ê°’

      // -1 1 -1 ... ìˆœìœ¼ë¡œ ì§„í–‰

    }

    return answer;
  }
}
